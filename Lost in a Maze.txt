    turn_left()

    if is_facing_north() and front_is_clear() and right_is_clear():
        move()
        turn_left()
    elif right_is_clear():
        turn_left()
        turn_left()
        turn_left()
        move()
        if right_is_clear() and not wall_in_front() and is_facing_north():
            build_wall()
        elif right_is_clear():
            turn_left()
            turn_left()
            turn_left()
            move()
        if front_is_clear():
            move()
    elif front_is_clear():
        move()
    elif wall_in_front() and wall_on_right():
